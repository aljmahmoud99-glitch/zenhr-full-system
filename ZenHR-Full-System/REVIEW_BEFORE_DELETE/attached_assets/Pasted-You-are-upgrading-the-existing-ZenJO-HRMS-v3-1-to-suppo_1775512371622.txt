PART 2 — TENANT ARCHITECTURE DECISION
═══════════════════════════════════════════════════════════════

Use: SHARED DATABASE + SHARED SCHEMA (most practical for this stack)
- Single MySQL database: zenjo
- All tables already have CompanyId column
- Tenant isolation via CompanyId filter on every query
- JWT token contains CompanyId claim

Why NOT separate databases per tenant:
- Replit environment — single MySQL socket
- Simpler migration
- Companies already exist as a table

TENANT IDENTIFICATION FLOW:
  User logs in → JWT issued with { userId, companyId, role }
                                              ↑
                                    This is the TenantId
  Every API call → Middleware extracts companyId from JWT
  Every DB query → filters by that companyId automatically

public class TenantService : ITenantService
{
    private readonly IHttpContextAccessor _http;

    public TenantService(IHttpContextAccessor http)
    {
        _http = http;
    }

    public int GetCompanyId()
    {
        var claim = _http.HttpContext?.User?.FindFirst("companyId")?.Value;
        if (int.TryParse(claim, out var id)) return id;
        throw new UnauthorizedAccessException("CompanyId claim missing from token");
    }

    public string GetUserRole()
        => _http.HttpContext?.User?.FindFirst(ClaimTypes.Role)?.Value ?? "";

    public int GetUserId()
    {
        var claim = _http.HttpContext?.User?.FindFirst("userId")?.Value;
        if (int.TryParse(claim, out var id)) return id;
        throw new UnauthorizedAccessException("UserId claim missing from token");
    }

    public bool IsSuperAdmin()
        => GetUserRole() == "SuperAdmin";
}

--- 4.6 CREATE: Tenant Validation Middleware ---
File: Middleware/TenantValidationMiddleware.cs

Validates that the companyId in JWT belongs to an active, non-expired company:

public class TenantValidationMiddleware
{
    private readonly RequestDelegate _next;

    public TenantValidationMiddleware(RequestDelegate next)
    {
        _next = next;
    }

    public async Task InvokeAsync(HttpContext context, AppDbContext db)
    {
        // Skip for auth endpoints
        if (context.Request.Path.StartsWithSegments("/api/auth") ||
            context.Request.Path.StartsWithSegments("/api/register"))
        {
            await _next(context);
            return;
        }

        // Skip if not authenticated
        if (!context.User.Identity?.IsAuthenticated ?? true)
        {
            await _next(context);
            return;
        }

        // SuperAdmin bypasses tenant validation
        if (context.User.IsInRole("SuperAdmin"))
        {
            await _next(context);
            return;
        }

        // Extract companyId from JWT
        var companyIdClaim = context.User.FindFirst("companyId")?.Value;
        if (!int.TryParse(companyIdClaim, out var companyId))
        {
            context.Response.StatusCode = 401;
            await context.Response.WriteAsJsonAsync(new { error = "Invalid tenant" });
            return;
        }

        // Validate company is active
        var company = await db.Companies
            .FirstOrDefaultAsync(c => c.Id == companyId && c.IsActive && !c.IsDeleted);

        if (company == null)
        {
            context.Response.StatusCode = 403;
            await context.Response.WriteAsJsonAsync(new {
                error = "الشركة غير نشطة أو غير موجودة",
                code = "TENANT_INACTIVE"
            });
            return;
        }

        // Check subscription expiry
        if (company.PlanExpiryDate.HasValue && company.PlanExpiryDate < DateOnly.FromDateTime(DateTime.Today))
        {
            context.Response.StatusCode = 402;
            await context.Response.WriteAsJsonAsync(new {
                error = "انتهت صلاحية اشتراك شركتك. يرجى التواصل مع الإدارة.",
                code = "SUBSCRIPTION_EXPIRED"
            });
            return;
        }

        await _next(context);
    }
}

--- 5.1 UPDATE: AuthService — store companyId from JWT ---
File: core/services/auth.service.ts

After successful login, decode JWT and store companyId:

interface JwtPayload {
  userId: number;
  companyId: number;   // ← add this
  role: string;
  name: string;
  email: string;
  exp: number;
}

export class AuthService {
  private currentUser$ = new BehaviorSubject<JwtPayload | null>(null);

  login(credentials: {username: string, password: string}): Observable<any> {
    return this.http.post<{accessToken: string, refreshToken: string}>(
      '/api/auth/login', credentials
    ).pipe(
      tap(response => {
        localStorage.setItem('accessToken', response.accessToken);
        localStorage.setItem('refreshToken', response.refreshToken);

        // Decode JWT to get user info including companyId
        const payload = this.decodeToken(response.accessToken);
        this.currentUser$.next(payload);
      })
    );
  }

  private decodeToken(token: string): JwtPayload {
    const base64 = token.split('.')[1];
    const decoded = JSON.parse(atob(base64));
    return {
      userId:    decoded['userId'],
      companyId: decoded['companyId'],  // ← extract
      role:      decoded['http://schemas.microsoft.com/ws/2008/06/identity/claims/role'],
      name:      decoded['http://schemas.xmlsoap.org/ws/2005/05/identity/claims/name'],
      email:     decoded['http://schemas.xmlsoap.org/ws/2005/05/identity/claims/emailaddress'],
      exp:       decoded['exp']
    };
  }

  get companyId(): number {
    return this.currentUser$.value?.companyId ?? 0;
  }

  get currentCompany$(): Observable<JwtPayload | null> {
    return this.currentUser$.asObservable();
  }
}
